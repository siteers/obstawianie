const app = initializeApp(firebaseConfig);
const db = getFirestore(app);
const table = document.getElementById("rankingTable");

let allSessions = [];

onSnapshot(collection(db, "sessions"), (snapshot) => {
    allSessions = [];
    snapshot.forEach(docu => {
        allSessions.push({ id: docu.id, ...docu.data() });
    });
    renderTable();
    renderCharts();
});

function renderTable() {
    table.innerHTML = `
        <tr>
            <th>Data</th>
            <th>Miejsce 1</th>
            <th>Miejsce 2</th>
            <th>Miejsce 3</th>
            <th>Akcje</th>
        </tr>`;
    allSessions.sort((a,b) => new Date(b.date) - new Date(a.date));
    allSessions.forEach(session => {
        table.innerHTML += `
        <tr>
            <td>${new Date(session.date).toLocaleString("pl-PL")}</td>
            <td>${session.results[0] || ""}</td>
            <td>${session.results[1] || ""}</td>
            <td>${session.results[2] || ""}</td>
            <td>
                <button class="editBtn" onclick="editSession('${session.id}')">Edytuj</button>
                <button class="deleteBtn" onclick="deleteSession('${session.id}')">Usuń</button>
            </td>
        </tr>`;
    });
}

window.deleteSession = async function(id) {
    if(confirm("Na pewno chcesz usunąć ten rekord?")) {
        await deleteDoc(doc(db, "sessions", id));
    }
}

window.editSession = async function(id) {
    const session = allSessions.find(s => s.id===id);
    if(!session) return;
    let newResults = prompt("Podaj nowe wyniki w kolejności 1,2,3 oddzielone przecinkiem", session.results.join(","));
    if(newResults) {
        newResults = newResults.split(",").map(s => s.trim());
        await db.collection("sessions").doc(id).update({ results: newResults });
    }
}

// ---- CHARTS ----
function renderCharts() {
    const ctxBar = document.getElementById("barChart").getContext("2d");
    const ctxPie = document.getElementById("pieChart").getContext("2d");

    // Liczba zwycięstw każdego imienia
    const winnerCount = {};
    allSessions.forEach(s => {
        const winner = s.results[0];
        if(winner) winnerCount[winner] = (winnerCount[winner]||0)+1;
    });

    const labels = Object.keys(winnerCount);
    const data = Object.values(winnerCount);

    // Bar chart
    new Chart(ctxBar, {
        type: 'bar',
        data: { labels, datasets:[{ label: "Liczba zwycięstw", data, backgroundColor:"#4d79ff" }] },
        options: { responsive:true, maintainAspectRatio:false }
    });

    // Pie chart
    new Chart(ctxPie, {
        type: 'pie',
        data: { labels, datasets:[{ label: "Zwycięstwa", data, backgroundColor:["#ff6384","#36a2eb","#ffce56","#4d79ff","#00cc44"] }] },
        options: { responsive:true, maintainAspectRatio:false }
    });
}
